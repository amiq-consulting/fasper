042        * @param _act The activation descriptor.
043        * @param _values The values for which to create instance.
044        * @return Returns a ParserActivation instance.
045        * @throws FasperException Is thrown if it couldn't create the desired instance.
046        */
047       ParserActivation getActivationFor(ActivationDescriptor _act
048          , Object[] _values) throws FasperException {
049          _act.setValues(_values);
050          return getActivationFor(_act);
051       }

053       /**
054        * Return a parser activation class
055        * @param _act The activation descriptor.
056        * @return Returns a parser activation class.
057        * @throws FasperException  Is thrown if it couldn't create the desired instance.
058        */
059       ParserActivation getActivationFor(ActivationDescriptor _act) 
060            throws FasperException {
061          String[] _sargs = _act.getArgs();
062          Class[] _args = new Class[_sargs.length];
063          for (int i = 0; i <= _sargs.length - 1; i++) {
064             try {
065                _args[i] = Class.forName(_sargs[i]);           
066             }catch(ClassNotFoundException _cnfe) { 
067                throw new FasperException(
068                   "UNKNOWN_ARGUMNET_CLASS"
069                   , "Can not create argument class : " + _sargs[i]
070                      + " for activation :" + _act.getName()
071                   , _cnfe);
072             };
073          };
074           try {
075             Class _class = Class.forName(_act.getName());
076             Constructor _cons = _class.getConstructor(_args);
077             ParserActivation _ac = (ParserActivation) _cons.newInstance(_act.getValues());
078             return _ac;
079          } catch(ClassNotFoundException _cnfe) {
080             throw new FasperException(
081                "UNKNOWN_ACTIVATION_CLASS"
082                , "Can not create instance of " + _act.getName());
083          } catch (NoSuchMethodException _nsme) {
084             throw new FasperException(
085                "UNKNOWN_ACTIVATION_CONSTRUCTOR"
086                , "Could not find a constructor for " + _act.getName());
087          } catch(ClassCastException _cce) {
088             throw new FasperException(
089                "WRONG_ACTIVATION_CLASS"
090                , "Class " + _act.getName() + " should implement " 
091                + "fasper.base.TestActivation or " 
092                + "fasper.base.ParserActivation");
093          } catch (Exception _ex) {
094             throw new FasperException(
095                "ACTIVATION_INSTANCE_FAILED"
096                , "Could not get an instance of " + _act.getName());
097          }
098       }

100       /**
101        * Creates a filter instance.
102        * @param _filt The full name of the filter class.
103        * @return Returns a filter instance of the specified type.
104        * @throws FasperException  
105        */
106       ASTFilter getFilter(String _filt) throws FasperException {
107          if (_filt == null) 
108             return new ASTFilter() {
109                    public String filterNode(AST _node){
110                       return _node.getText();
111                }
112                    public boolean isNodeIgnored(AST _node) {
113                       return false;
114                    }
115                    public void useTokenMap(HashMap _map){
116                    }
117             };
118          try {
119             Class _class = Class.forName(_filt);
120             Constructor _cons = _class.getConstructor(new Class[0]);
121             ASTFilter _ac = (ASTFilter) _cons.newInstance(new Object[0]);
122             return _ac;
123          } catch(ClassNotFoundException _cnfe) {
124             throw new FasperException(
125                "UNKNOWN_FILTER_CLASS"
126                , "Can not create instance of " + _filt);
127          } catch (NoSuchMethodException _nsme) {
128             throw new FasperException(
129                "UNKNOWN_FILTER_CONSTRUCTOR"
130                , "Could not find a constructor for " + _filt);
131          } catch (ClassCastException _cce) {
132             throw new FasperException(
133                "WRONG_FILTER_CLASS"
134                , "Class " + _filt + " should implement fasper.base.FilterBank");
135          } catch (Exception _ex) {
136             throw new FasperException(
137                "FILTER_INSTANCE_FAILED"
138                , "Could not get an instance of " + _filt);
139          }
140       }
