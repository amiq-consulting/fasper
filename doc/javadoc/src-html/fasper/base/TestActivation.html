001    package fasper.base;
002    
003    import java.util.*;
004    /**
005     * Common interface for test activations.
006     * It extends the functionality of the ParserActivation with a method 
007     * for retrieving the parser under test activation descriptor. 
008     * @see fasper.gm.GMActivation
009     * @see fasper.base.ParserActivation 
010     */
011    public interface TestActivation extends ParserActivation {
012       
013       /**
014        * Returns an activation descriptor of the parser under test.
015        * @return <code>ActivationDescriptor</code> activation descriptor 
016        * of the parser under test.
017        */
018       public ActivationDescriptor getPADescriptor();
019       
020    }
