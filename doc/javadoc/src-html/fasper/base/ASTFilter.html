001    package fasper.base;
002    
003    import antlr.collections.*;
004    import java.util.*;
005    /**
006     * Common interface for token text filters.
007     * See examples in this package's description.
008     * @see fasper.DefaultASTComparator
009     */
010    public interface ASTFilter {
011       /**
012        * Filters the input string of the specified AST node. 
013        * You can use some regex to filter the strings (token's text). 
014        * Ussualy filters that inherit this interface are put in a HashMap and mapped
015        * by token type string (eg SUPER_TOKEN). 
016        * @param _node the string to be filtered
017        * @return Returns the filtered string
018        */
019       public String filterNode(AST _node);
020       
021       /**
022        * Computes ignore state of the given node.
023        * Returns true if node is ignored (and also it's children).
024        * @param _node The node to be filtered.
025        * @return Returns <code>true</code> if <code>_node</code> and it's kids 
026        * should be ignored  
027        */
028       public boolean isNodeIgnored(AST _node);
029       
030       /**
031        * Specify the tokenMap to be used. 
032        * @param _tokenMap
033        */
034       public void useTokenMap(HashMap _tokenMap);
035       
036    }
