001    package fasper.base;
002    
003    import antlr.collections.AST;
004    import java.util.*;
005    
006    /**
007     * Comparator common interface.
008     * It should be inherited by any AST comparator in order to be plugin-able.
009     * @see fasper.DefaultASTComparator
010     * @see fasper.base.ASTFilter
011     * @see fasper.gm.GMActivation
012     * @see fasper.gm
013     */
014    public interface ASTComparator {
015       
016       /**
017        * Entry point for AST comparision process.
018        * @param expected is the expected AST (the golden model).
019        * @param resulted is the AST corresponding to the target source.
020        * @throws Throwable which is the result of the comparision.
021        * @see fasper.DefaultASTComparator
022        * @see fasper.FasperException
023        * @see fasper.ComparisonException
024        */
025       public void compare(AST expected, AST resulted) throws Throwable;
026       
027       /**
028        * Specify the filter bank to be used.
029        * There are situations when you would like to filter out some nodes or
030        * ignore whole branches of the tree. Theese are handled by an ASTFilter.
031        * @param _filter to be used by this comparator should inherit 
032        * from ASTFilter
033        * @see fasper.base.ASTFilter
034        */
035       public void useFilter(ASTFilter _filter);
036       
037       /**
038        * Specify the tokenMap to be used. 
039        * @see fasper.gm 
040        * @see fasper.DefaultASTComparator
041        * @see fasper.Factory
042        * @param _tokenMap
043        */
044       public void useTokenMap(HashMap _tokenMap);
045    }
