001    package fasper.base;
002    
003    /**
004     * Makes implementation of string filters very easy. All you need to do is to
005     * create a annonymous class like :<br>
006     * 
007     * <PRE>
008     * StringFilter one = new StringFilter() { 
009          public String filter(String input) {
010          String res = input.replaceAll("\\\\\n", ""); 
011          res = res.replaceAll("[ \\s]+","").trim(); 
012          return res; 
013          } 
014       };
015       // and filter a string based on <code>one</code> filter.
016       String _mda = one.filter("sadsasd     dadsasdsa    dassads");
017     * <PRE>
018     * 
019     * @see fasper.base description
020     */
021    public interface StringFilter {
022    
023       /**
024        * Filters a string.
025        * @param _in A string to be filtered
026        * @return Returns the filtered string.
027        */
028       public String filter(String _in);
029    
030    }
