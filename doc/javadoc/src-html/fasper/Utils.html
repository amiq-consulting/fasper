001    package fasper;
002    
003    import java.io.*;
004    import java.net.URL;
005    import javax.swing.ImageIcon;
006    
007    import fasper.base.*;
008    
009    
010    import java.util.*;
011    
012    /**
013     * Contains utility methods.
014     */
015    class Utils {
016       static {
017          FasperException.addErrorCode("DEFAULTS_LOAD_FAILED");
018       }
019       
020       /**
021        * Logs messages to system.out.
022        * @param _s The message.
023        */
024       static void log(String _s) {
025          System.out.println(_s);
026       }
027    
028       /**
029        * Loads a file of properties.
030        * @param _file The file handle to load.
031        * @return Returns the properties read from file.
032        * @throws FasperException
033        */
034       static Properties loadProps(File _file) throws FasperException {
035          Properties properties = new Properties();
036          try {
037             properties.load(new FileInputStream(_file));
038             return properties;
039          } catch (IOException e) {
040             throw new FasperException("DEFAULTS_LOAD_FAILED"
041                , "Error while loading properties file : " + e.toString());
042          }
043       }
044       
045       static String[] tokenize(String input) {
046          ArrayList v = new ArrayList();
047          StringTokenizer t = new StringTokenizer(input);
048          
049          while (t.hasMoreTokens())
050             v.add(t.nextToken());
051          return (String[]) v.toArray(new String[v.size()]);
052       }
053       
054       /**
055        * Returns the extension of a file.
056        * @param f The file handle.
057        * @return Returns the extension of the file.
058        */
059       public static String getExtension(File f) {
060          String ext = null;
061          String s = f.getName();
062          int i = s.lastIndexOf('.');
063          if (i > 0 &&  i < s.length() - 1)
064             ext = s.substring(i+1).toLowerCase();
065          return ext;
066       }
