001    package fasper;
002    
003    import java.io.*;
004    
005    import fasper.base.*;
006    
007    import antlr.collections.*;
008    
009    /**
010     * A proxy for the Central class.
011     * This class should be used in tests.
012     */
013    public class Tool {
014       private Defs defs = Defs.getMe();
015       private Central central = new Central();
016       
017       /**
018        * A basic constructor.
019        */
020       public Tool() {
021       }
022       
023       /**
024        * Loads a source or test file.
025        * @param _file A source or test file.
026        * @throws Throwable is thrown if errors occured during 
027        * load process
028        */
029       public void load(File _file) throws Throwable {
030          if (_file == null)
031             throw new FasperException("FILE_IS_NULL"
032                , "The file you are tring to load is null");
033          if (_file.getName().endsWith(defs.TEST_EXT))
034             central.loadTest(_file);
035          else
036             central.loadFile(_file);
037       }
038       
039       /**
040        * Loads a source or test file.
041        * @param _file is the name of the file to be loaded
042        * @throws Throwable is thrown if errors occured during 
043        * load process
044        */
045       public void load(String _file) throws Throwable {
046          this.load(new File(_file));
047       }
048       
049       /**
050        * Loads a default values file.
051        * @param _file the default values file.
052        * @see fasper.Defs
053        * @throws Throwable throws an Exception if could not load
054        * defaults
055        */
056       public void loadDefaults(String _file) throws Throwable {
057          central.loadDefaults(_file);
058       }
059       
060       /**
061        * Saves the current verifies AST in z-file
062        * @param _file the z-file.
063        * @throws Throwable throws an Exception if couldn't save 
064        * the test.
065        */
066       public void saveTest(File _file) throws Throwable {
067          central.saveTest(_file);
068       }
069       
070       /**
071        * Sets the AST filter class name (eg. woo.test.WooFilter)
072        * @param _filt
073        */
074       public void setFilter(String _filt) {
075          defs.FILTER = _filt;
076       }
077       
078       /**
079        * Sets the comparator class name (eg. woo.test.WooComparator)
080        * @param _comp the name of the comparator class
081        */
082       public void setComparator(String _comp) {
083          defs.COMP_CLASS = _comp;
084       }
085       
086       /**
087        * Returns current expected AST or null if there isn't one.
088        * @return AST current expected AST
089        */
090       public AST getExpectedAST() {
091          return central.getExpectedAST();
092       }
093       
094       /**
095        * Returns current parser AST or null if there isn't one.
096        * @return AST current parser AST
097        */
098       public AST getParserAST() {
099          return central.getParserAST();
100       }
101       
102       /**
103        * Returns current treeParser AST or null if there isn't one.
104        * @return AST current treeParser AST
105        */
106       public AST getTreeParserAST() {
107          return central.getTreeParserAST();
108       }
109       
110       /**
111        * 
112        * @return AST currently verified AST
113        */
114       public AST getVerifiedAST() {
115          return central.getVerifiedAST();
116       }   
117    }
