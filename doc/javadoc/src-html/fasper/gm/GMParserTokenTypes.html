009    /**
010     * GMParser is used for parsing the test files. The role of a test file is to 
011     * keep golden model (expected tree) & activation in one place.
012     * In this case a test file has next format:
013     <code>
014    dutActivationClass myparser.MyParserActivation<br>
015    argTypeAndValue java.lang.String #"e_0.e"#<br>
016    --><br>
017    [C_PROGRAM, #"top: a_c.c"#]<br>
018    |--[C_MODULE, #"module: a_c.c"#]<br>
019    |  |  |--[LITERAL_struct, #"struct"#]<br>
020    |  |  |  |--[ID, #"s_1"#]<br>
021    |  |  |--[LITERAL_struct, #"struct"#]<br>
022    |  |  |  |--[ID, #"s_2"#]<br>
023     </code>
024     On the short :<br>
025     - strings are enclosed in <b>#"</b><i>a_string</i><b>"#</b><br>
026     - <b>dutActivationClass</b> : activates the parser under test. This class must 
027     implement fasper.base.Activation<br>
028     - <b>argTypeAndValue</b> : pair type-value. The type must be used with 
029     full path eg. java.lang.String.<br>
030       As you can see the structure of the golden model tree is very similar to <br>
031    the one displayed by the ASTFrame.
032     * @author stefan@amiq.ro
033     */
034    
035    public interface GMParserTokenTypes {
036            int EOF = 1;
037            int NULL_TREE_LOOKAHEAD = 3;
038            int START = 4;
039            int TEST_LIST = 5;
040            int TEST = 6;
041            int OUT = 7;
042            int LITERAL_dutActivationClass = 8;
043            int ID = 9;
044            int LITERAL_argTypeAndValue = 10;
045            int STRING = 11;
046            int STARTGM = 12;
047            int ENDGM = 13;
048            int VLINE = 14;
049            int LBRACK = 15;
050            int COMMA = 16;
051            int RBRACK = 17;
052    }
