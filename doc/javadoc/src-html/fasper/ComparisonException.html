001    package fasper;
002    
003    /**
004     * ComparisonException
005     */
006    public class ComparisonException extends FasperException {
007       static { 
008          FasperException.addErrorCode("NOT_USED");
009          FasperException.addErrorCode("PASSED");
010          FasperException.addErrorCode("EXPECTED_AST_IS_NULL");
011          FasperException.addErrorCode("RESULTED_AST_IS_NULL");
012          FasperException.addErrorCode("BOTH_ASTS_ARE_NULL");
013          FasperException.addErrorCode("EXPECTED_NODE_IS_NULL");
014          FasperException.addErrorCode("RESULTED_NODE_IS_NULL");
015          FasperException.addErrorCode("BOTH_NODES_ARE_NULL");
016          FasperException.addErrorCode("TOKEN_TEXT_MISMATCH");
017          FasperException.addErrorCode("TOKEN_TYPE_MISMATCH");
018          FasperException.addErrorCode("DIFFERENT_AST_STRUCTURE");  
019       }
020       
021       private String context = "";
022    
023       public ComparisonException() {
024          super();
025       }
026    
027       /**
028        * @param _mesaj
029        */
030       public ComparisonException(String _mesaj) {
031          super(_mesaj);
032       }
033    
034       /**
035        * @param _mesaj
036        * @param _cause
037        */
038       public ComparisonException(String _mesaj, Throwable _cause) {
039          super(_mesaj, _cause);
040       }
041    
042       /**
043        * @param errorCode
044        * @param message
045        */
046       public ComparisonException(String errorCode, String message) {
047          super(errorCode, message);
048       }
049    
050       /**
051        * @param errorCode
052        * @param message
053        * @param _source
054        */
055       public ComparisonException(String errorCode, String message,
056          Exception _source) {
057          super(errorCode, message, _source);
058       }
059       
060       public String toString() {
061          if (this.getErrorCode().equals("PASSED"))
062             return "***Fasper: AST Comparision Status = PASSED" 
063                    + "\nContext : " + context;
064          String _result = "*** Error: " + getMessage() + "."
065             + "\n" + "    [see " + getErrorCode() + " spec for more details]";
066          _result += "Context : " + context;
067          if (this.getCause() != null)
068             _result += "\n" + "    Cause : " + this.getCause().toString();
069          return _result;
070       }
071    
072       
073       public String getContext() {
074          return context;
075       }
076       
077       public void setContext(String context) {
078          this.context = context;
079       }
080    }
