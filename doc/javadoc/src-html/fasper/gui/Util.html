001    package fasper.gui;
002    
003    import java.io.*;
004    import java.net.URL;
005    import javax.swing.ImageIcon;
006    
007    import fasper.Tool;
008    
009    import java.util.*;
010    
011    class Util {
012    
013       static ImageIcon getImageIcon(String path) {
014          URL imgURL = Tool.class.getResource(path);
015          if (imgURL != null)
016             return new ImageIcon(imgURL);
017          Util.log("Icon " + path  + " can not be created!");
018          return null;
019       }
020       
021       static void log(String _s) {
022          System.out.println(_s);
023       }
024       
025       static String getExtension(File f) {
026          String ext = null;
027          String s = f.getName();
028          int i = s.lastIndexOf('.');
029          if (i > 0 &&  i < s.length() - 1)
030             ext = s.substring(i+1).toLowerCase();
031          return ext;
032       }
033       
034       static String getPathFrom(String _name, File _file) {
035          if (_file != null)
036             return _file.getParent() + File.separator + _name; 
037          return _name;
038       } 
039    }
