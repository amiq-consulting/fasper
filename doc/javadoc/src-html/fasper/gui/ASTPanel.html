001    package fasper.gui;
002    
003    import javax.swing.*;
004    
005    import java.awt.*;
006    import antlr.collections.*;
007    
008    class ASTPanel extends JPanel {
009       ASTSplitPanel parent;
010       private JTree tree;
011       private JScrollPane scroller = new JScrollPane();
012       
013       ASTPanel(ASTSplitPanel _parent, AST _ast) {
014          super(new BorderLayout());
015          parent = _parent;
016          refresh(_ast);
017       }
018       
019       ASTPanel(ASTSplitPanel _parent) {
020          this(_parent, null);
021       }
022       
023       JScrollPane getScroller() {
024          return scroller;
025       }
026       
027       void refresh(AST _ast) {
028          this.removeAll();
029          if (_ast != null) {
030             tree = new ASTree(this, _ast);
031             scroller = new JScrollPane(tree);
032             this.add(scroller, BorderLayout.CENTER);
033          } else
034             add(new JLabel("AST is not available!"
035                , Util.getImageIcon("resources/warning.gif")
036                , JLabel.HORIZONTAL), BorderLayout.CENTER);
037          repaint();
038       }
039       
040       void syncScrolling(boolean _sync){
041          parent.syncScrolling(_sync);
042       }
043    }
