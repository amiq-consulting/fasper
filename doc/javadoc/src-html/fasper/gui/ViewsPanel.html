001    package fasper.gui;
002    
003    import javax.swing.*;
004    
005    import fasper.*;
006    
007    class ViewsPanel extends JTabbedPane {
008       private SourceViewer source = new SourceViewer();
009       private ASTSplitPanel erPanel = new ASTSplitPanel();
010       private ASTSplitPanel pwPanel = new ASTSplitPanel();
011       private ModelHub model;
012       
013       ViewsPanel(ModelHub _model) {
014          super();
015          this.model = _model;
016          model.setViews(this);
017          erPanel.setLeftAST("Expected AST", new ASTPanel(erPanel));
018          erPanel.setRightAST("Resulted AST", new ASTPanel(erPanel));
019          pwPanel.setLeftAST("Parser AST", new ASTPanel(pwPanel));
020          pwPanel.setRightAST("TreeParser AST", new ASTPanel(pwPanel));
021          addTab("E-R AST"
022             , Util.getImageIcon("resources/e_r_ast.gif")
023             , erPanel 
024             ,  "Show Expected and Resulted ASTs");
025          addTab("P-W AST"
026             , Util.getImageIcon("resources/p_w_ast.gif")
027             , pwPanel
028             , "Shows Parser and TreeParser ASTs");
029          addTab("Source"
030             , Util.getImageIcon("resources/source.gif")
031             , source.getViewer()
032             , "Shows Source File");
033       }
034       
035       void refresh()  {
036          erPanel.refresh(model.getTool().getExpectedAST()
037             , model.getTool().getVerifiedAST());
038          pwPanel.refresh(model.getTool().getParserAST()
039             , model.getTool().getTreeParserAST());
040          source.refresh(model.lastFile);
041       }
042    }
