001    package fasper.gui;
002    
003    import javax.swing.*;
004    
005    import fasper.*;
006    
007    import java.awt.*;
008    import java.awt.event.*;
009    
010    public class Fasper extends JFrame {
011       private static Tool tool = new Tool();
012       private static ModelHub model;
013       private static Console console;
014       private static BaseMenuBar menu = null;
015       private static Fasper frame;
016       private Defs defs = Defs.getMe();
017    
018       Fasper() {
019          super("Fasper");
020          setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
021          addWindowListener(new WindowAdapter() {
022             public void windowClosing(WindowEvent evt) {
023                System.out.println("\nFasper exits . . .");
024                System.exit(0);
025             }
026          });
027          setSize(defs.WIDTH, defs.HEIGHT);
028          setResizable(true);
029          model = new ModelHub(this, tool);
030          initComponents();
031       }
032    
033       private void initComponents() {
034          ConsolePanel _console = new ConsolePanel(model, console);
035          JSplitPane _jspv = new JSplitPane(JSplitPane.VERTICAL_SPLIT);
036          _jspv.setResizeWeight(.25D);
037          _jspv.setOneTouchExpandable(true);
038          _jspv.setDividerLocation(2 * defs.HEIGHT / 3);
039          _jspv.setPreferredSize(new Dimension(defs.WIDTH, defs.HEIGHT));
040          ViewsPanel views = new ViewsPanel(model);
041          _jspv.setTopComponent(views);
042          _jspv.setBottomComponent(_console);
043          getContentPane().add(_jspv, BorderLayout.CENTER);
044          menu = new BaseMenuBar(this, model);
045       }
046    
047       public static void main(String[] args) {
048          String _test = "";
049          String _defs = null;
050          String _filt = null;
051          String _comp = null;
052          int _j = 0;
053          for (int _i = 0; _i < args.length; _i++) {
054             _j = args[_i].split("=").length;
055             if (args[_i].startsWith("-test") && _j == 2)
056                _test = args[_i].split("=")[1].trim();
057             else if (args[_i].startsWith("-defs") && _j == 2)
058                _defs = args[_i].split("=")[1].trim();
059             else if (args[_i].startsWith("-filter") && _j == 2)
060                _filt = args[_i].split("=")[1].trim();
061             else if (args[_i].startsWith("-comp") && _j == 2)
062                _comp = args[_i].split("=")[1].trim();
063             else
064                usage();
065          }
066          if (_defs != null) 
067             try {
068                tool.loadDefaults(_defs);
069             } catch (Throwable _thr) {
070                Util.log("Could not load defaults file " + _defs + " because of : \n"
071                   + _thr.toString());
072             }
073          if (_filt != null)
074             tool.setFilter(_filt);
075          if (_comp != null)
076             tool.setComparator(_comp);
077          try {
078             console = new Console();
079          } catch (Exception e) {
080             System.out.println(e.getMessage());
081             System.exit(0);
082          }
083          initGui(_test);
084       }
085    
086       private static void initGui(String _test) {
087          frame = new Fasper();
088          Runnable runner = new FrameThread(frame);
089          EventQueue.invokeLater(runner);
090          if (!_test.equals(""))
091             model.load(_test);
092       }
093    
094       static void setWorkState(boolean _ste) {
095          if (_ste) {
096             menu.setToolBarEnable(false);
097             frame.setCursor(Cursor.getPredefinedCursor(Cursor.WAIT_CURSOR));
098          } else {
099             menu.setToolBarEnable(true);
100             frame.setCursor(Cursor.getPredefinedCursor(Cursor.DEFAULT_CURSOR));
101          }
102          frame.validate();
103       }
104    
105       private static void usage() {
106          System.out.println("-------\nUsage : \n" + "Runs Fasper in gui mode.\n"
107             + "java fasper.gui.Fasper[ -test=test.z][ -defs=a.defs]"
108             + "[ -filter=my.pack.AFilter][ -comp=my.pack.AComparator]\n"
109             + "-test=<test.z>   : specify the test/file to be loaded\n"
110             + "-defs=<defs>     : specify the properties file to be loaded\n"
111             + "-filter=<filter> : specify the filter class\n"
112             + "-comp=<comp>     : specify the comparator class\n" + "\nExample :"
113             + "java fasper.gui.Fasper -test=gica.z -filter=mda.AFilter "
114             + "-comp=mda.AComparator ");
115       }
116    
117       private static class FrameThread implements Runnable {
118          private final JFrame frame;
119    
120          FrameThread(JFrame frame) {
121             this.frame = frame;
122          }
123    
124          public void run() {
125             frame.pack();
126             frame.setVisible(true);
127          }
128       }
129    }
